### 드래그 리사이즈

```js
scwin.onpageload = function() {
            com.foldingLayout(); // 공통 접기 실행
            scwin.initDragResize();
        };

        scwin.initDragResize = function() {
            // 1. [핵심] 현재 내 화면 프레임의 ID를 구하여 ID 셀렉터를 생성합니다.
            var pId = $p.getFrame().id;
            var prefix = (pId === "mf") ? "" : "#" + pId + " ";
            
            // 2. 현재 내 메뉴 페이지 안의 요소들만 정확하게 선택합니다.
            var wrapper = document.querySelector(prefix + '.lybox_wrap.foldingbox');
            var left = wrapper ? wrapper.querySelector('.lycolumn:first-child') : null;
            var handle = document.querySelector(prefix + '.btn_resize');
            var btnFolding = document.querySelector(prefix + '.btn_folding');
            
            if (!wrapper || !left || !handle) return;

            // 3. [접기 연동] 내 페이지의 접기 버튼에만 이벤트를 한정하여 바인딩
            if (btnFolding && !btnFolding.dataset.init) {
                btnFolding.dataset.init = "true"; // 중복 바인딩 방지
                btnFolding.addEventListener('click', function() {
                    setTimeout(function() {
                        left.style.flex = wrapper.classList.contains('on') ? '' : '0 0 0';
                    }, 50);
                });
            }

            // 4. [드래그 조절] 마우스 다운 시 해당 화면 내에서만 작동하도록 제어
            handle.onmousedown = function(e) {
                if (!wrapper.classList.contains('on')) return; // 접힌 상태면 무시
                e.preventDefault();
                
                var startX = e.clientX;
                var startWidth = left.offsetWidth;
                var wrapperWidth = wrapper.offsetWidth;
                
                document.body.style.userSelect = 'none';
                
                // 드래그 중
                document.onmousemove = function(moveEvent) {
                    var newWidth = startWidth + (moveEvent.clientX - startX);
                    if (newWidth > 100 && newWidth < wrapperWidth - 100) {
                        left.style.flex = '0 0 ' + newWidth + 'px';
                    }
                };
                
                // 마우스 뗌 (이벤트 완전 삭제)
                document.onmouseup = function() {
                    document.onmousemove = null;
                    document.onmouseup = null;
                    document.body.style.userSelect = '';
                };
            };
        };

```
