### 그리드.. 데이터에 따라 커스텀포맷팅

```html
	<!-- body column에 customFormatter 연결 -->
	<w2:column blockSelect="false" id="" inputType="text" width="100" customFormatter="scwin.fnCusFormat"></w2:column>
```
```js

/*************************************************************
 * Event: scwin.fnCusFormat
 * 기 능: 상태 값별 컬러 포맷
 ************************************************************/
scwin.fnCusFormat = function (data, formattedData, rowIndex, colIndex) {
	var joinSituNm = data ;
	var cellClassMap = {
		"승인": "bold-text color-type02 bg-color-type02",
	};
  
	var cellClass = cellClassMap[joinSituNm] || "";

	grd2.setCellClass(rowIndex, colIndex, cellClass);

	return data;
};

```
