### 이미지 슬라이더가 깨졌던 이유

```js
/**************************************************************************
 * 함 수 명: scwin.setImageViewer
 * 기 능: 이미지 미리보기 화면 호출
 **************************************************************************/
scwin.setImageViewer = function(arvAflId) {
  ajaxLib.ajax("/cf/CFCommon/getAttachFileList", {
    mode		: "asynchronous",
    method 		: "POST",
    requestData	: {docId: arvAflId},
    errCallback	: "scwin.fnErrCallback",
    callback	: function(result) {
      const resultList = result.data.dsAttachFileList;
      const $track = $("#" + grpImageViewer.getID()).find('.image-track');

      $track.empty();

      if (!resultList || resultList.length === 0) {
          return;
      }

      resultList.sort((a, b) => Number(a.docSeq) - Number(b.docSeq));
      dsFileList.setJSON(resultList);

      resultList
          .filter(function(file) {
              const ext = file.extsNm?.toLowerCase().trim();
              return ['jpg', 'jpeg', 'gif', 'bmp', 'tiff', 'png', 'webp'].includes(ext);
          })
          .forEach(function(file) {
              const url = `${download.url}${download.groupId}${file.docId}${download.seq}${file.docSeq}`;
              const $slide = $('<div>').addClass('image-slide');
              const $img   = $('<img>').attr('src', url).attr('alt', file.docNm || '').addClass('image');
              $slide.append($img);
              $track.append($slide);
          });

      setTimeout(function() {
          com.setupImageSlider(grpImageViewer.getID());
      }, 100);
    }
  });
};

```

- 같은 image-track 안에서 WebSquare 컴포넌트(xf:image)와 순수 HTML(img)를 섞은 게 충돌 포인트

- 슬라이더 로직은 .image-slide > img 구조를 기준으로 동작하는데, xf:image는 런타임에 WebSquare 방식으로 렌더링되어 DOM 구조/라이프사이클이 다릅니다.

그래서 초기화 타이밍, 이벤트 바인딩, slide 개수 계산이 꼬이기 쉽습니다.

좋은 질문입니다. 핵심은 누가 그 DOM의 생명주기를 관리하느냐입니다.

관리 주체가 다름

WebSquare 컴포넌트 DOM: WebSquare 엔진이 생성, 갱신, 제거를 관리

raw DOM: 브라우저와 jQuery가 직접 관리

렌더 구조가 다름

xf:image를 선언해도 화면에는 엔진이 변환한 구조로 나옴

중간 래퍼 요소, 내부 속성, 컴포넌트 상태 노드가 붙을 수 있음

그래서 개발자가 기대한 순수 img 구조와 달라질 수 있음

이벤트 바인딩 경로가 다름

WebSquare는 ev:onclick, 컴포넌트 이벤트 체계 사용

raw DOM은 addEventListener, jQuery on 사용

둘을 섞으면 이벤트가 중복되거나 한쪽만 동작하는 일이 생김

재렌더 시점이 다름

WebSquare가 화면 갱신할 때 컴포넌트 영역 DOM을 다시 그림

그 안에 수동으로 append한 raw DOM이 사라지거나 순서가 바뀔 수 있음

이번 슬라이더에 왜 영향이 큰가

슬라이더 로직은 image-slide > img 같은 특정 구조를 전제로 계산

같은 트랙 안에 xf:image와 raw img가 섞이면 개수 계산, transform 대상, 클릭 바인딩이 꼬이기 쉬움

결론:
섞는 것 자체가 절대 금지는 아니지만, 같은 컨테이너에서는 매우 취약합니다.
실무에서는 한 컨테이너 한 방식으로 고정하는 게 가장 안전합니다.
