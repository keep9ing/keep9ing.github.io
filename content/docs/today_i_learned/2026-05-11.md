    /**
     * 함수명: scwin._fetchWeather
     * 기  능: 기상청·에어코리아 API 4개를 Promise.all 로 병렬 호출한 뒤
     *         renderWeather 에 결과를 전달합니다.
     *         - getVilageFcst  : 단기예보 (오늘~D+2, 시간대별 기온·하늘·강수)
     *         - getMidTa       : 중기기온 (D+3~D+7 최고/최저)
     *         - getMidLandFcst : 중기육상 (D+3~D+7 날씨 상태 문자열)
     *         - getCtprvnRltmMesureDnsty : 실시간 미세먼지 (Nginx 프록시 경유)
     * @param {number} nx     기상청 격자 X 좌표
     * @param {number} ny     기상청 격자 Y 좌표
     * @param {object} region 지역 객체 (REGIONS 배열 원소)
     */
    /**
     * 함수명: scwin._fetchWeather
    * 기  능: 백엔드에 기상청 3건 + 에어코리아 1건을
    *         단일 POST 요청으로 위임합니다. API 키는 서버에서만 관리됩니다.
     * @param {number} nx     기상청 격자 X 좌표
     * @param {number} ny     기상청 격자 Y 좌표
     * @param {object} region 지역 객체 (REGIONS 배열 원소)
     */
    scwin._fetchWeather = function (nx, ny, region) {
        var now = new Date();
        var base    = scwin._getKmaBase(now);
        var midTmFc = scwin.getMidTmFc(now);
        var body = JSON.stringify({
            nx:           String(nx),
            ny:           String(ny),
            baseDate:     base.date,
            baseTime:     base.time,
            midTmFc:      midTmFc,
            midTaRegId:   region.midTaRegId,
            midLandRegId: region.midLandRegId,
            sidoName:     region.sidoName
        });

        var weatherApiUrl = window.location.pathname.indexOf('/sc/') === 0 ? '/sc/ep/Main/getWeatherData' : '/ep/Main/getWeatherData';

        fetch(weatherApiUrl, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: body
        })
            .then(function (r) { return r.json(); })
            .then(function (res) {
                var payload    = (res && res.data) ? ((res.data && res.data.data) ? res.data.data : res.data) : (res || {});
                var shortItems = scwin._toItems(payload.shortFcst);
                var midTa      = scwin._toItems(payload.midTa)[0] || null;
                var midLand    = scwin._toItems(payload.midLandFcst)[0] || null;
                var dustData   = scwin._extractDustData(payload.dustInfo);
                if (!shortItems || shortItems.length === 0) {
                    scwin._weatherLoading('예보 데이터가 없습니다.');
                    return;
                }
                scwin.renderWeather(shortItems, midTa, midLand, now, region.cityName, dustData);
            }).catch(function () {
                scwin._weatherLoading('날씨 정보를 불러올 수 없습니다.');
            });
    };


    /* **************************************************************************
     * [5] 시각 계산 유틸
     *     API 호출에 필요한 기준시각(base_date, base_time, tmFc)을 계산합니다.
     * ************************************************************************** */

    /**
     * 함수명: scwin._pad
     * 기  능: 숫자를 2자리 0-패딩 문자열로 변환합니다. (예: 9 → '09')
     */
    scwin._pad = function (n) { return String(n).padStart(2, '0'); };

    /**
     * 함수명: scwin._getKmaBase
     * 기  능: 단기예보 API의 base_date · base_time 을 계산합니다.
     *         발표 시각: 02·05·08·11·14·17·20·23시 (발표 후 10분부터 조회 가능)
     *         현재 시각이 02시 10분 이전이면 전일 23시 기준으로 폴백합니다.
     * @param  {Date}   now 기준 시각
     * @return {{ date: string, time: string }}  예) { date: '20260508', time: '1100' }
     */
    scwin._getKmaBase = function (now) {
        var baseTimes = [2, 5, 8, 11, 14, 17, 20, 23];
        var h = now.getHours(), m = now.getMinutes(), base = -1;
        for (var i = baseTimes.length - 1; i >= 0; i--) {
            if (h > baseTimes[i] || (h === baseTimes[i] && m >= 10)) { base = baseTimes[i]; break; }
        }
        var d = new Date(now);
        if (base === -1) { d.setDate(d.getDate() - 1); base = 23; }  // 전일 23시 기준
        return {
            date: d.getFullYear() + scwin._pad(d.getMonth() + 1) + scwin._pad(d.getDate()),
            time: scwin._pad(base) + '00'
        };
    };

    /**
     * 함수명: scwin.getMidTmFc
     * 기  능: 중기예보 API의 tmFc 파라미터 값을 계산합니다.
     *         발표 기준: 18시 이후 → 당일 1800 / 06시 이후 → 당일 0600 / 그 외 → 전일 1800
     * @param  {Date}   now 기준 시각
     * @return {string} 'YYYYMMDD0600' 또는 'YYYYMMDD1800' 형식 문자열
     */
    scwin.getMidTmFc = function (now) {
        var h = now.getHours(), d = new Date(now);
        var hhmm = h >= 18 ? '1800' : h >= 6 ? '0600' : (d.setDate(d.getDate() - 1), '1800');
        return d.getFullYear() + scwin._pad(d.getMonth() + 1) + scwin._pad(d.getDate()) + hhmm;
    };


    /* **************************************************************************
     * [6] 응답 파싱 유틸
     *     API JSON 응답에서 필요한 데이터를 안전하게 추출합니다.
     * ************************************************************************** */

    /**
     * 함수명: scwin._toItems
     * 기  능: 공공데이터포털 표준 응답(response.body.items.item)을
     *         항상 배열로 반환합니다. 응답 구조가 없거나 단일 객체인 경우도 처리합니다.
     */
    scwin._toItems = function (response) {
        if (!response) { return []; }

        // 백엔드 정규화 응답: 바로 배열 또는 { item: [...] }
        if (Array.isArray(response)) { return response; }
        if (Array.isArray(response.item)) { return response.item; }
        if (response.item && typeof response.item === 'object') { return [response.item]; }

        // 공공데이터 원본 응답: response.body.items.item
        var items = (((response.response || {}).body || {}).items || {}).item;
        if (Array.isArray(items)) { return items; }
        if (items && typeof items === 'object') { return [items]; }
        return [];
    };

    /**
     * 함수명: scwin._extractDustData
     * 기  능: 에어코리아 API 응답에서 첫 번째 측정소 데이터를 추출합니다.
     *         응답 구조가 공공데이터포털 표준과 다르므로 별도로 처리합니다.
     */
    scwin._extractDustData = function (dustResponse) {
        if (!dustResponse) { return null; }

        var list = [];

        // 백엔드 정규화 응답
        if (Array.isArray(dustResponse)) {
            list = dustResponse;
        } else if (dustResponse.item && Array.isArray(dustResponse.item)) {
            list = dustResponse.item;
        } else if (dustResponse.item && typeof dustResponse.item === 'object') {
            list = [dustResponse.item];
        }

        // 공공데이터 원본 응답
        if (list.length === 0) {
            var dustItems = ((dustResponse.response || {}).body || {}).items;
            if (Array.isArray(dustItems)) {
                list = dustItems;
            } else if (dustItems && Array.isArray(dustItems.item)) {
                list = dustItems.item;
            } else if (dustItems && dustItems.item && typeof dustItems.item === 'object') {
                list = [dustItems.item];
            }
        }

        if (!list || list.length === 0) { return null; }

        // pm10/pm25 값이 숫자인 측정소를 우선 사용 (값 '-' 회피)
        var best = list[0];
        var bestScore = -1;
        for (var i = 0; i < list.length; i++) {
            var row = list[i] || {};
            var pm10 = row.pm10Value != null ? row.pm10Value : row.pm10Value24;
            var pm25 = row.pm25Value != null ? row.pm25Value : row.pm25Value24;
            var score = 0;
            if (!isNaN(parseInt(pm10, 10))) { score++; }
            if (!isNaN(parseInt(pm25, 10))) { score++; }
            if (score > bestScore) {
                best = row;
                bestScore = score;
                if (bestScore === 2) { break; }
            }
        }

        return best;
    };

     * [7] 렌더링 오케스트레이터
     *     API 응답을 받아 위젯 전체를 갱신하는 진입점입니다.

    /**
     * 함수명: scwin.renderWeather
     * 기  능: 날씨 API 응답 4종을 받아 위젯 DOM 전체를 업데이트합니다.
     *         ① 단기예보 정규화 → ② 오늘 패널 갱신 → ③ 주간 행 갱신 → ④ 미세먼지 패널 갱신
     * @param {Array}  shortItems 단기예보 아이템 배열 (getVilageFcst)
     * @param {object} midTa      중기기온 예보 객체 (getMidTa, 없으면 null)
     * @param {object} midLand    중기육상 예보 객체 (getMidLandFcst, 없으면 null)
     * @param {Date}   now        API 호출 기준 시각
     * @param {string} cityName   표시할 지역명
     * @param {object} dustData   미세먼지 측정 데이터 (없으면 null)
     */
    scwin.renderWeather = function (shortItems, midTa, midLand, now, cityName, dustData) {
        var days      = scwin._collectShortDays(shortItems);   // 단기예보 → 날짜별 버킷
        var shortKeys = Object.keys(days).sort();              // 날짜 오름차순 정렬
        var midCards  = scwin._buildMidCards(shortKeys, midTa, midLand, now);  // D+3~D+7 중기카드

        var tk = shortKeys[0];
        if (!tk) { scwin._weatherLoading('예보 데이터가 없습니다.'); return; }

        scwin._updateTodayPanel(tk, days[tk], cityName);       // 오늘 패널 DOM 갱신
        scwin._renderWeeklyRows(shortKeys, days, midCards, tk); // 주간 행 DOM 갱신
        scwin._renderDustPanel(dustData);                      // 미세먼지 패널 DOM 갱신

        $('#wLoading').hide();                                 // 로딩 메시지 숨김
        $('#wIconHuge, #wCityLabel, .w-main-area, .w-dust, .w-temp-minmax').show();  // 오늘 패널 표시
        $('.w-weekly').show();    // 주간 예보 영역 표시
    };


    /* **************************************************************************
     * [8] 렌더링 보조 함수
     *     renderWeather 가 호출하는 개별 DOM 갱신 및 데이터 가공 함수들입니다.
     * ************************************************************************** */

    /**
     * 함수명: scwin._collectShortDays
     * 기  능: 단기예보 item 배열을 날짜(fcstDate)별 버킷 객체로 변환합니다.
     *         각 날짜 버킷에는 TMP·TMN·TMX·SKY·PTY·POP 배열이 포함됩니다.
     */
    scwin._collectShortDays = function (shortItems) {
        var days = {};
        shortItems.forEach(function (item) {
            var d = item.fcstDate;
            if (!days[d]) {
                days[d] = { TMP: [], TMN: null, TMX: null, SKY: [], PTY: [], POP: [] };
            }
            if (item.category === 'TMP') days[d].TMP.push(Number(item.fcstValue));
            if (item.category === 'TMN') days[d].TMN = Number(item.fcstValue);
            if (item.category === 'TMX') days[d].TMX = Number(item.fcstValue);
            if (item.category === 'SKY') days[d].SKY.push(item.fcstValue);
            if (item.category === 'PTY') days[d].PTY.push(item.fcstValue);
            if (item.category === 'POP') days[d].POP.push(Number(item.fcstValue));
        });
        return days;
    };

    /**
     * 함수명: scwin._buildMidCards
     * 기  능: 단기예보에 포함되지 않은 D+3~D+7 날짜에 대해
     *         중기예보 데이터로 카드 객체 배열을 생성합니다.
     */
    scwin._buildMidCards = function (shortKeys, midTa, midLand, now) {
        var midCards = [];
        // 중기 API는 D+4~D+10 필드(taMin4/wf4Am ...)를 제공합니다.
        for (var i = 4; i <= 7; i++) {
            var dt = new Date(now);
            dt.setDate(dt.getDate() + i);
            var ds = dt.getFullYear() + scwin._pad(dt.getMonth() + 1) + scwin._pad(dt.getDate());
            if (shortKeys.indexOf(ds) === -1) {
                midCards.push({
                    dateStr : ds,
                    minTemp : (midTa && midTa['taMin' + i] != null) ? midTa['taMin' + i] : '--',
                    maxTemp : (midTa && midTa['taMax' + i] != null) ? midTa['taMax' + i] : '--',
                    wf      : (midLand && (midLand['wf' + i + 'Pm'] || midLand['wf' + i + 'Am']))
                                ? (midLand['wf' + i + 'Pm'] || midLand['wf' + i + 'Am'])
                                : '맑음'
                });
            }
        }
        return midCards;
    };

    /**
     * 함수명: scwin._updateTodayPanel
     * 기  능: 오늘 날씨 패널(아이콘·지역명·날짜·기온)을 DOM에 반영합니다.
     */
    scwin._updateTodayPanel = function (todayKey, todayData, cityName) {
        var t    = scwin._tempRange(todayData);
        var tMid = Math.floor(todayData.SKY.length / 2);
        var tSky = todayData.SKY[tMid] || '1';
        var tPty = todayData.PTY[tMid] || '0';

        $('#wIconHuge').attr('class', 'w-icon-huge ' + scwin.getWeatherIcon(tSky, tPty));
        $('#wCityLabel').text(cityName);
        $('.w-city-dropdown-item').removeClass('active').filter('[data-city="' + cityName + '"]').addClass('active');
        $('#wDescDate').text(scwin._getMmdd(todayKey) + ' (' + scwin._getDow(todayKey) + ')');
        $('#wTempHuge').text(t.max + '°');
        $('#wTempMax').text(t.max + '°');
        $('#wTempMin').text(t.min + '°');
    };

    /**
     * 함수명: scwin._updateWeekRow
     * 기  능: 주간 예보 한 행(날짜·아이콘·설명·최고/최저)을 DOM에 반영합니다.
     *         #wWeek{idx}Date / Icon / Desc / Max / Min 요소를 갱신합니다.
     * @param {number} idx     0-based 행 인덱스
     * @param {string} dateKey 'YYYYMMDD' 형식 날짜
     */
    scwin._updateWeekRow = function (idx, dateKey, icon, desc, maxT, minT) {
        var p = '#wWeek' + idx;
        var maxText = (maxT == null || maxT === '') ? '--' : maxT;
        var minText = (minT == null || minT === '') ? '--' : minT;
        var descText = (desc == null || desc === '') ? '-' : desc;
        $(p + 'Date').text(scwin._getMmdd(dateKey) + ' (' + scwin._getDow(dateKey) + ')');
        $(p + 'Icon').attr('class', 'w-wr-icon ' + icon);
        $(p + 'Desc').text(descText);
        $(p + 'Max').text(maxText + '°');
        $(p + 'Min').text(minText + '°');
    };

    /**
     * 함수명: scwin._renderWeeklyRows
     * 기  능: 단기예보(D+1~) 행과 중기예보(D+3~D+7) 행을 순서대로 렌더링합니다.
     *         단기 날짜는 오늘(shortKeys[0])을 제외한 나머지를 사용합니다.
     */
    scwin._renderWeeklyRows = function (shortKeys, days, midCards, todayKey) {
        var midByDate = {};
        var i;
        for (i = 0; i < midCards.length; i++) {
            midByDate[midCards[i].dateStr] = midCards[i];
        }

        function addDays(dateStr, plusDays) {
            var y = parseInt(dateStr.substring(0, 4), 10);
            var m = parseInt(dateStr.substring(4, 6), 10) - 1;
            var d = parseInt(dateStr.substring(6, 8), 10);
            var dt = new Date(y, m, d);
            dt.setDate(dt.getDate() + plusDays);
            return dt.getFullYear() + scwin._pad(dt.getMonth() + 1) + scwin._pad(dt.getDate());
        }

        // 화면 고정 7칸: 내일(D+1)~D+7을 날짜 순서대로 항상 채웁니다.
        for (i = 1; i <= 7; i++) {
            var dateKey = addDays(todayKey, i);
            var d = days[dateKey];
            var mc = midByDate[dateKey];
            if (d) {
                var mi  = Math.floor(d.SKY.length / 2);
                var t   = scwin._tempRange(d);
                var sky = d.SKY[mi] || '1';
                var pty = d.PTY[mi] || '0';
                scwin._updateWeekRow(i - 1, dateKey, scwin.getWeatherIcon(sky, pty), scwin.getWeatherDesc(sky, pty), t.max, t.min);
            } else if (mc) {
                scwin._updateWeekRow(i - 1, dateKey, scwin.getMidWeatherIcon(mc.wf), mc.wf, mc.maxTemp, mc.minTemp);
            } else {
                scwin._updateWeekRow(i - 1, dateKey, 'wi-sunny', '-', '--', '--');
            }
        }
    };

    /**
     * 함수명: scwin._renderDustPanel
     * 기  능: 미세먼지(PM10)·초미세먼지(PM2.5) 수치와 등급을 DOM에 반영합니다.
     *         dustData 가 null 이면 아무것도 갱신하지 않습니다.
     */
    scwin._renderDustPanel = function (dustData) {
        if (!dustData) { return; }
        var pm10Value = dustData.pm10Value != null ? dustData.pm10Value : dustData.pm10Value24;
        var pm25Value = dustData.pm25Value != null ? dustData.pm25Value : dustData.pm25Value24;
        var pm10Grade = scwin._getDustGrade('pm10', pm10Value);
        var pm25Grade = scwin._getDustGrade('pm25', pm25Value);
        $('#wPm10Val').text((pm10Value || '--') + 'μg');
        $('#wPm10Grade').text(pm10Grade.label).attr('class', 'w-dust-grade ' + pm10Grade.class);
        $('#wPm25Val').text((pm25Value || '--') + 'μg');
        $('#wPm25Grade').text(pm25Grade.label).attr('class', 'w-dust-grade ' + pm25Grade.class);
    };

    /**
     * 함수명: scwin._weatherLoading
     * 기  능: 로딩 오버레이 메시지를 표시합니다.
     *         renderWeather 완료 시 $('#wLoading').hide() 로 숨깁니다.
     */
    scwin._weatherLoading = function (msg) {
        $('#wLoading').text(msg).show();
        $('#wIconHuge, .w-main-area, .w-dust, .w-temp-minmax').hide();
    };


    /* **************************************************************************
     * [9] 날씨 코드 변환 함수
     *     코드/문자열 → 아이콘·설명·등급 변환 순수 함수들입니다.
     * ************************************************************************** */

    /**
     * 함수명: scwin.getWeatherIcon
     * 기  능: SKY(하늘) / PTY(강수) 코드를 아이콘 CSS 클래스명으로 변환합니다.
     *         PTY가 '0' 이 아니면(강수 있음) PTY 우선 적용합니다.
     */
    scwin.getWeatherIcon = function (sky, pty) {
        var w = scwin._WEATHER;
        return (w.pty[pty] || w.sky[sky] || w.def)[0];
    };

    /**
     * 함수명: scwin.getWeatherDesc
     * 기  능: SKY / PTY 코드를 한글 날씨 설명 문자열로 변환합니다.
     */
    scwin.getWeatherDesc = function (sky, pty) {
        var w = scwin._WEATHER;
        return (w.pty[pty] || w.sky[sky] || w.def)[1];
    };

    /**
     * 함수명: scwin.getMidWeatherIcon
     * 기  능: 중기예보 날씨 문자열(wf)을 아이콘 CSS 클래스명으로 변환합니다.
     *         _MID_WF_ICON_MAP 순서대로 문자열 포함 여부를 검사합니다.
     */
    scwin.getMidWeatherIcon = function (wf) {
        if (!wf) { return 'wi-sunny'; }
        for (var i = 0; i < scwin._MID_WF_ICON_MAP.length; i++) {
            if (wf.indexOf(scwin._MID_WF_ICON_MAP[i][0]) > -1) {
                return scwin._MID_WF_ICON_MAP[i][1];
            }
        }
        return 'wi-sunny';
    };

    /**
     * 함수명: scwin._getDustGrade
     * 기  능: 미세먼지 수치를 등급 레이블 및 CSS 클래스로 변환합니다.
     *         _DUST_THRESHOLDS / _DUST_LABELS 상수를 참조합니다.
     * @param {string} type  'pm10' 또는 'pm25'
     * @param {string|number} value 측정값
     * @return {{ label: string, class: string }}
     */
    scwin._getDustGrade = function (type, value) {
        var v = parseInt(value, 10);
        if (isNaN(v)) { return { label: '--', class: '' }; }
        var thresholds = scwin._DUST_THRESHOLDS[type] || scwin._DUST_THRESHOLDS.pm25;
        if (v <= thresholds[0]) return scwin._DUST_LABELS[0];  // 좋음
        if (v <= thresholds[1]) return scwin._DUST_LABELS[1];  // 보통
        if (v <= thresholds[2]) return scwin._DUST_LABELS[2];  // 나쁨
        return scwin._DUST_LABELS[3];                          // 매우나쁨
    };


    /* **************************************************************************
     * [10] 지역 유틸
     *      REGIONS 배열 조회 헬퍼 함수들입니다.
     * ************************************************************************** */

    /** REGIONS 배열의 첫 번째 항목(기본 지역)을 반환합니다. */
    scwin._getDefaultRegion = function () {
        return scwin.REGIONS[0] || null;
    };

    /**
     * cityName 으로 REGIONS 배열에서 지역 객체를 검색합니다.
     * 일치하는 항목이 없으면 null 을 반환합니다.
     */
    scwin._findRegionByName = function (cityName) {
        return scwin.REGIONS.filter(function (r) { return r.cityName === cityName; })[0] || null;
    };


    /* **************************************************************************
     * [11] 날짜 / 포맷 유틸
     *      'YYYYMMDD' 형식 문자열을 표시용 문자열로 변환합니다.
     * ************************************************************************** */

    /**
     * 함수명: scwin._getDow
     * 기  능: 'YYYYMMDD' → 요일 한글 (예: '20260508' → '금')
     */
    scwin._getDow = function (dateStr) {
        var y = parseInt(dateStr.substring(0, 4), 10);
        var m = parseInt(dateStr.substring(4, 6), 10) - 1;
        var d = parseInt(dateStr.substring(6, 8), 10);
        return scwin._DOW[new Date(y, m, d).getDay()];
    };

    /**
     * 함수명: scwin._getMmdd
     * 기  능: 'YYYYMMDD' → 'M/D' 형식 문자열 (예: '20260508' → '5/8')
     */
    scwin._getMmdd = function (dateStr) {
        return parseInt(dateStr.substring(4, 6), 10) + '/' + parseInt(dateStr.substring(6, 8), 10);
    };

    /**
     * 함수명: scwin._tempRange
     * 기  능: 단기예보 하루 버킷에서 최저/최고 기온을 반환합니다.
     *         TMN·TMX(일최저·최고) 값이 있으면 우선 사용하고,
     *         없으면 TMP(시간대별 기온) 배열의 min·max 로 산출합니다.
     * @return {{ min: number|string, max: number|string }}
     */
    scwin._tempRange = function (d) {
        return {
            min: d.TMN !== null ? Math.round(d.TMN) : (d.TMP.length ? Math.round(Math.min.apply(null, d.TMP)) : '--'),
            max: d.TMX !== null ? Math.round(d.TMX) : (d.TMP.length ? Math.round(Math.max.apply(null, d.TMP)) : '--')
        };
    };


    /* **************************************************************************
     * [12] 날씨 이벤트 핸들러
     * ************************************************************************** */

    /**
     * 함수명: scwin.onCitySelectChange
     * 기  능: 지역 드롭다운에서 항목을 선택했을 때 호출됩니다.
     *         선택된 cityName 으로 지역 객체를 찾아 날씨를 재조회합니다.
     */
    scwin.onCitySelectChange = function (cityName) {
        var region = scwin._findRegionByName(cityName);
        if (region) {
            scwin._weatherLoading(cityName + ' 날씨를 불러오는 중...');
            scwin._fetchWeather(region.nx, region.ny, region);
        }
    };

    /**
     * 함수명: scwin.btnWeatherDetail_onclick
     * 기  능: 기상청 날씨 상세 페이지를 새 탭으로 엽니다.
     */
    scwin.btnWeatherDetail_onclick = function () {
        window.open('https://www.weather.go.kr/w/index.do', '_blank');
    };

    private Object parseApiResponse(String urlStr, ObjectMapper mapper) throws Exception {
        HttpURLConnection conn = openConnection(urlStr);
        conn.setRequestMethod("GET");
        conn.setConnectTimeout(10000);
        conn.setReadTimeout(10000);

        int statusCode = conn.getResponseCode();
        InputStream responseStream = statusCode >= 400 ? conn.getErrorStream() : conn.getInputStream();
        if (responseStream == null) {
            responseStream = new ByteArrayInputStream(new byte[0]);
        }

        try (BufferedReader reader = new BufferedReader(new InputStreamReader(
            responseStream, "UTF-8"))) {
            StringBuilder sb = new StringBuilder();
            String line;
            while ((line = reader.readLine()) != null) {
                sb.append(line);
            }

            if (statusCode >= 400) {
                log.warn("weather api error status: {}, url: {}, body: {}", statusCode, urlStr, sb.toString());
                throw new RestException(HttpStatus.BAD_GATEWAY,
                        "외부 날씨 API 호출에 실패했습니다. (status=" + statusCode + ")");
            }

            JsonNode root = mapper.readTree(sb.toString());
            JsonNode itemsNode = root.path("response").path("body").path("items");
            JsonNode dataNode = itemsNode.path("item");

            // 기상청: items.item, 에어코리아: items(배열) 모두 지원
            if (dataNode.isMissingNode() || dataNode.isNull()) {
                dataNode = itemsNode;
            }
            if (dataNode.isMissingNode() || dataNode.isNull()) return null;
            return mapper.convertValue(dataNode, Object.class);
        } finally {
            conn.disconnect();
        }
    }

    private HttpURLConnection openConnection(String urlStr) throws Exception {
        HttpURLConnection conn = (HttpURLConnection) new URL(urlStr).openConnection();

        // 로컬 환경에서만 SSL 검증 우회 (호출 단위로만 적용)
        if ("local".equals(activeProfile) && conn instanceof HttpsURLConnection) {
            TrustManager[] trustAll = new TrustManager[] { new X509TrustManager() {
                public X509Certificate[] getAcceptedIssuers() { return null; }
                public void checkClientTrusted(X509Certificate[] certs, String authType) {}
                public void checkServerTrusted(X509Certificate[] certs, String authType) {}
            } };
            SSLContext sc = SSLContext.getInstance("TLS");
            sc.init(null, trustAll, new java.security.SecureRandom());

            HttpsURLConnection httpsConn = (HttpsURLConnection) conn;
            httpsConn.setSSLSocketFactory(sc.getSocketFactory());
            httpsConn.setHostnameVerifier((host, session) -> true);
        }

        return conn;
    }

    private String encode(String value) throws Exception {
        return URLEncoder.encode(value == null ? "" : value, "UTF-8");
    }
